761TDSLOGIN*
762tds_init_login(TDSLOGIN *login, TDSLOCALE * locale)
763{
764 char hostname[128];
765#if HAVE_NL_LANGINFO1 && defined(CODESETCODESET)
766 const char *charset;
767#else
768 char *lc_all, *tok = NULL((void*)0);
769#endif
770
771 /*
772 * TDS 7.0:
773 * 0x02 indicates ODBC driver
774 * 0x01 means change to initial language must succeed
775 */
776 login->option_flag2 = 0x03;
777 login->tds_version = TDS_DEFAULT_VERSION0x500;
778 login->block_size = 0;
779
780#if HAVE_NL_LANGINFO1 && defined(CODESETCODESET)
781 charset = nl_langinfo(CODESETCODESET);

783 charset = "ISO-8859-1";
784 if (!tds_dstr_copy(&login->client_charset, charset))
785 return NULL((void*)0);
786#else
787 if (!tds_dstr_copy(&login->client_charset, "ISO-8859-1"))
788 return NULL((void*)0);
789
790 if ((lc_all = strdup(setlocale(LC_ALL__LC_ALL, NULL((void*)0)))) == NULL((void*)0))
791 return NULL((void*)0);
792
793 if (strtok_r(lc_all, ".", &tok)) {
794 char *encoding = strtok_r(NULL((void*)0), "@", &tok);
795#ifdef _WIN32
796 /* windows give numeric codepage*/
797 if (encoding && atoi(encoding) > 0) {
798 char *p;
799 if (asprintf(&p, "CP%s", encoding) >= 0) {
800 free(lc_all);
801 lc_all = encoding = p;
802 }
803 }
804#endif
805 if (encoding) {
806 if (!tds_dstr_copy(&login->client_charset, encoding))
807 return NULL((void*)0);
808 }
809 }
810 free(lc_all);
811#endif
812
813 if (locale) {
814 if (locale->language)
815 if (!tds_dstr_copy(&login->language, locale->language))
816 return NULL((void*)0);
817 if (locale->server_charset)
818 if (!tds_dstr_copy(&login->server_charset, locale->server_charset))
819 return NULL((void*)0);
820 }
821 if (tds_dstr_isempty(&login->language)((&login->language)->dstr_size == 0)) {
822 if (!tds_dstr_copy(&login->language, TDS_DEF_LANG"us_english"))
823 return NULL((void*)0);
824 }

826 gethostname(hostname, sizeof(hostname));
827 hostname[sizeof(hostname) - 1] = '\0'; /* make sure it's truncated */
828 if (!tds_dstr_copy(&login->client_host_name, hostname))
829 return NULL((void*)0);
830
831 return login;
832}
